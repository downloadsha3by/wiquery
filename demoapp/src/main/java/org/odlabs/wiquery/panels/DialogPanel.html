Dialog box example

This example show how to build a dialog box with WiQuery

Click me to open a dialog box with existing content
WiQuery rocks !

<a wicket:id="openSimpleDialog">Click me to open a dialog box with existing content</a> <div title="Simple dialog with WiQuery" wicket:id="simpleDialog"> WiQuery rocks ! </div>

public MyWebPage(String id) { super(id); Button button = new Button("openSimpleDialog"); this.add(button); final Dialog dialog = new Dialog("simpleDialog"); button.add(new WiQueryEventBehavior(new Event(MouseEvent.CLICK) { private static final long serialVersionUID = 1L; @Override public JsScope callback() { return JsScope.quickScope(dialog.open().render()); } })); this.add(dialog);

public MyWebPage(String id) { button = new Button("openAjaxDialog"); this.add(button); final Dialog ajaxDialog = new Dialog("ajaxDialog"); final WebMarkupContainer container = new WebMarkupContainer("ajaxContent"); container.setOutputMarkupId(true); ajaxDialog.add(container); button.add(new WiQueryAjaxEventBehavior(MouseEvent.CLICK) { private static final long serialVersionUID = 1L; @Override protected void onEvent(AjaxRequestTarget target) { if (newContent == null) { newContent = new AjaxContent("ajaxContent"); newContent.setOutputMarkupId(true); container.replaceWith(newContent); target.addComponent(newContent, container.getMarkupId()); } ajaxDialog.open(target); } }); this.add(ajaxDialog); }
